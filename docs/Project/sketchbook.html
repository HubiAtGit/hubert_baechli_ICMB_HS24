                               lt = LearningHouers,
                               lr = kb) { 
                                # lr = LearningRate (optional)
  tb <- ( 1 - kb )^( 1 / -lr )  # assumed time learnd allready
  ke <- 1 - ( tb + lt )^( -lr ) # Knowlege after time learnd
